name: Generate PR & Merge Documentation

on:
  pull_request:
    types:
      - opened
      - synchronize
      - reopened
      - closed

permissions:
  contents: write
  pull-requests: write

jobs:
  generate-documentation:
    if: github.event.action != 'closed' || github.event.pull_request.merged == true

    runs-on: ubuntu-latest

    env:
      GEMINI_API_KEY: ${{ secrets.GEMINI_API_KEY }}

    steps:
      - name: Checkout Repository
        uses: actions/checkout@v4
        with:
          fetch-depth: 0
          ref: ${{ github.event.action == 'closed' && github.event.pull_request.merge_commit_sha || github.head_ref }}

      - name: Setup Python
        uses: actions/setup-python@v5
        with:
          python-version: "3.12"

      - name: Install Dependencies
        run: |
          python -m pip install --upgrade pip
          pip install -r requirements.txt

      - name: Build Merge Context
        run: |
          python scripts/build_merge_context.py \
            --source-branch "feat/testlocal" \
            --target-branch "main" \
            --base-sha "bc07009d71ab7f3ac3aa6c4973ade7065cc9e1fa" \
            --merge-sha "25643b23c7cce6b71053180a7c625ead4f799267" \
            --output context.json

      - name: Generate Documentation
        run: |
          python src/generate_documentation.py \
            --context context.json \
            --output DOCUMENTATION.md \
            --use-gemini

      # -------------------------
      # BEFORE MERGE
      # -------------------------
      - name: Upload Documentation Artifact
        if: github.event.action != 'closed'
        uses: actions/upload-artifact@v4
        with:
          name: pr-documentation
          path: DOCUMENTATION.md

      # -------------------------
      # AFTER MERGE
      # -------------------------
      - name: Commit Documentation
        if: github.event.action == 'closed' && github.event.pull_request.merged == true
        run: |
          git config user.name "github-actions[bot]"
          git config user.email "41898282+github-actions[bot]@users.noreply.github.com"

          git fetch origin
          git checkout ${{ github.event.pull_request.base.ref }}
          git pull origin ${{ github.event.pull_request.base.ref }}

          git add DOCUMENTATION.md

          if git diff --cached --quiet; then
            echo "No documentation changes."
            exit 0
          fi

          git commit -m "docs: update DOCUMENTATION.md after merge of PR #${{ github.event.pull_request.number }}"

          git push origin ${{ github.event.pull_request.base.ref }}
